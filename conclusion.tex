\item[-] Data Partitioning
\item[-] Data Computation
\end{itemize}

\item Query Driven  Join
\begin{itemize}
\item[-] Query Decomposition
\item[-] Communication among Sub-Queries
\item[-] Combine results of Sub-Queries
\end{itemize}

\end{itemize}

\begin{frame}
\frametitle{Future Work}
\begin{itemize}
\item Enrich the Naive Bayes re-partitioning strategy for kNN stream join (theoretical and experimental analysis and proof --- Accuracy, Load Balance )

\item Enrich the SPARQL query engine by providing operations such as FILTER, OPTIONAL, etc.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Future Applications}
\begin{itemize}
\item Real Time Recommendation System Based on kNN

\item Real Time Natural Language Processing System Based on RDF
\end{itemize}
\end{frame}
